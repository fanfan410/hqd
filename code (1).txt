\section{Phát biểu bài toán}

\begin{frame}{1.1 Mô tả bài toán & Mục tiêu}
    \begin{itemize}
        \item Dựa trên bộ dữ liệu giao dịch bất động sản (nhà ở một gia đình) đã bán ở \textbf{Miami} (theo nguồn Kaggle). 
        \item \textbf{Mục tiêu:} Xây dựng mô hình học máy để dự đoán giá bán của ngôi nhà (\texttt{SALE\_PRC}).
        \item Sử dụng các thuộc tính liên quan như: diện tích đất, diện tích sống, khoảng cách đến các tiện ích, tuổi của ngôi nhà, chất lượng cấu trúc, v.v.
    \end{itemize}
\end{frame}

\begin{frame}{1.2 Đầu vào: Giới thiệu bộ dữ liệu}
    \textbf{Miami Housing Dataset}
    \begin{itemize}
        \item Nguồn dữ liệu: Kaggle (\url{https://www.kaggle.com/datasets/deepcontractor/miami-housing-dataset/data})
        \item Số lượng mẫu: 13,932 ngôi nhà
        \item Kích thước dữ liệu: 1.64 MB
        \item Năm dữ liệu: 2016
    \end{itemize}
\end{frame}

\begin{frame}{1.2 Đầu vào: Các thuộc tính (1/2)}
    Bộ dữ liệu ở định dạng CSV với các cột thông tin:
    \begin{itemize}
        \item \texttt{LATITUDE}: Vĩ độ của ngôi nhà.
        \item \texttt{LONGITUDE}: Kinh độ của ngôi nhà.
        \item \texttt{PARCELNO}: Số hiệu lô đất (có thể dùng làm định danh duy nhất).
        \item \textbf{\texttt{SALE\_PRC}}: Giá bán của ngôi nhà (\textbf{biến mục tiêu}).
        \item \texttt{LND\_SQFOOT}: Diện tích đất (đơn vị: feet vuông).
        \item \texttt{TOT\_LVG\_AREA}: Tổng diện tích sống (đơn vị: feet vuông).
        \item \texttt{SPEC\_FEAT\_VAL}: Giá trị các đặc điểm đặc biệt (ví dụ: hồ bơi, cảnh quan).
        \item \texttt{RAIL\_DIST}: Khoảng cách đến đường sắt (đơn vị: mét).
    \end{itemize}
\end{frame}

\begin{frame}{1.2 Đầu vào: Các thuộc tính (2/2)}
    \begin{itemize}
        \item \texttt{OCEAN\_DIST}: Khoảng cách đến đại dương (đơn vị: mét).
        \item \texttt{WATER\_DIST}: Khoảng cách đến nguồn nước gần nhất (đơn vị: mét).
        \item \texttt{CNTR\_DIST}: Khoảng cách đến trung tâm thành phố (đơn vị: mét).
        \item \texttt{SUBCNTR\_DI}: Khoảng cách đến trung tâm phụ (đơn vị: mét).
        \item \texttt{HWY\_DIST}: Khoảng cách đến đường cao tốc (đơn vị: mét).
        \item \texttt{age}: Tuổi của ngôi nhà (tính bằng năm).
        \item \texttt{avno60plus}: Số chuyến bay trên 60 decibel (có thể là biến nhị phân hoặc số nguyên).
        \item \texttt{month\_sold}: Tháng bán nhà (1-12).
        \item \texttt{structure\_quality}: Chất lượng cấu trúc (thang điểm từ 1-5).
    \end{itemize}
\end{frame}

\begin{frame}{1.3 Đầu ra & 1.4 Yêu cầu xử lý}
    \textbf{1.3 Đầu ra}
    \begin{itemize}
        \item Kết quả dự đoán: Giá bán dự đoán của ngôi nhà (\texttt{SALE\_PRC}) dưới dạng số thực.
    \end{itemize}
    \vspace{1em}
    \textbf{1.4 Yêu cầu xử lý}
    \begin{itemize}
        \item Tiền xử lý dữ liệu: Đảm bảo chất lượng đầu vào (xử lý giá trị thiếu, chuẩn hóa dữ liệu, mã hóa biến phân loại nếu cần).
        \item Lựa chọn và huấn luyện mô hình học máy phù hợp (ví dụ: hồi quy tuyến tính, rừng ngẫu nhiên, gradient boosting).
        \item Đánh giá hiệu suất mô hình: Sử dụng các chỉ số như MSE, RMSE, hoặc R².
        \item Tối ưu hóa mô hình nếu cần (điều chỉnh siêu tham số, thử nghiệm các mô hình khác).
    \end{itemize}
\end{frame}

\begin{frame}{2.4 Tiền xử lý: Xử lý Outliers (IQR)}
    \begin{itemize}
        \item Dùng IQR (Interquartile Range) để phát hiện và loại bỏ outlier trong biến mục tiêu \texttt{SALE\_PRC}.
        \item Tính Q1 (phân vị 25\%), Q3 (phân vị 75\%), IQR = Q3 - Q1.
        \item Xác định giới hạn:
            \begin{itemize}
                \item \texttt{lower\_bound = Q1 - 1.5 * IQR}
                \item \texttt{upper\_bound = Q3 + 1.5 * IQR}
            \end{itemize}
        \item Giới hạn các giá trị của \texttt{SALE\_PRC} vào khoảng [\texttt{lower\_bound}, \texttt{upper\_bound}] bằng phương thức \texttt{clip()}, tạo cột mới \texttt{SALE\_PRC\_CLEANED}.
    \end{itemize}

\begin{frame}{2.4 Tiền xử lý: Chuẩn hóa, Mã hóa, Loại bỏ cột}
    \textbf{Chuẩn Hóa Dữ Liệu (Standardization)}
    \begin{itemize}
        \item Các cột số như \texttt{LND\_SQFOOT}, \texttt{TOT\_LVG\_AREA}, \texttt{RAIL\_DIST}, etc., có đơn vị và phạm vi khác nhau, cần chuẩn hóa.
        \item Ví dụ: Dùng \texttt{StandardScaler} từ \texttt{sklearn.preprocessing} để đưa về phân phối chuẩn có mean=0, std=1.
    \end{itemize}
    \vspace{0.5em}

\section{Tạo, luyện và đánh giá mô hình}

\begin{frame}{3.1 Lựa chọn mô hình & Quyết định}
    Trong bài toán dự đoán giá nhà, các mô hình hồi quy sau được thử nghiệm:
    \begin{itemize}
        \item \textbf{Linear Regression:} Mô hình hồi quy tuyến tính cơ bản, làm baseline.
        \item \textbf{Ridge Regression:} Hồi quy tuyến tính với Regularization L2 (giảm overfitting, xử lý đa cộng tuyến).
        \item \textbf{Lasso Regression:} Hồi quy tuyến tính với Regularization L1 (giảm overfitting, tự động lựa chọn đặc trưng).
        \item \textbf{Random Forest Regressor:} Mô hình Ensemble (nhiều cây quyết định), tốt cho dữ liệu phi tuyến, ít nhạy cảm với outlier.
        \item \textbf{Gradient Boosting Regressor:} Mô hình Ensemble nâng cao (tuần tự sửa lỗi), thường cho hiệu suất rất tốt.
    \end{itemize}
    \textbf{Lý do lựa chọn:} Dựa trên bản chất bài toán hồi quy, tính chất thường phi tuyến của dữ liệu bất động sản, và mong muốn so sánh các kỹ thuật từ cơ bản đến nâng cao.
\end{frame}

\begin{frame}{3.2 Luyện mô hình: Đánh giá ban đầu (CV)}
    \begin{itemize}
        \item Sử dụng K-Fold Cross-Validation (K=5) trên tập huấn luyện để đánh giá hiệu suất ban đầu của các mô hình với tham số mặc định.
        \item Chỉ số đánh giá chính: RMSE (Root Mean Squared Error).
    \end{itemize}
    \begin{center}
        \includegraphics[width=0.8\textwidth]{fig10_boxplot_rmse_cv.png} \\ % <<-- CẦN FILE ẢNH
        \tiny{Hình 10: So sánh RMSE của các mô hình qua Cross-Validation}
    \end{center}
    \textbf{Nhận xét:}
    \begin{itemize}
        \item Mô hình Gradient Boosting cho kết quả RMSE trung bình thấp nhất và độ phân tán (độ dài hộp) cũng tương đối nhỏ, cho thấy hiệu suất tốt và ổn định.
        \item Linear, Ridge, Lasso có RMSE cao hơn đáng kể. Random Forest tốt hơn nhưng chưa bằng Gradient Boosting.
        \item => Chọn Gradient Boosting để tiếp tục tinh chỉnh siêu tham số.
    \end{itemize}
\end{frame}

\begin{frame}{3.2 Luyện mô hình: Tinh chỉnh (GridSearchCV)}
    \begin{itemize}
        \item Chọn Gradient Boosting làm mô hình chính.
        \item Tiến hành tối ưu hóa siêu tham số bằng \textbf{GridSearchCV} để tìm tổ hợp tốt nhất.
        \item \textbf{Không gian siêu tham số tìm kiếm (param\_grid):}
            \begin{itemize}
                \item \texttt{n\_estimators}: [100, 200, 300] (Số lượng cây)
                \item \texttt{learning\_rate}: [0.01, 0.05, 0.1] (Tốc độ học)
                \item \texttt{max\_depth}: [3, 5, 7] (Độ sâu tối đa của cây)
                \item \texttt{min\_samples\_split}: [2, 5, 10] (Số mẫu tối thiểu để chia nhánh)
                \item \texttt{subsample}: [0.8, 0.9, 1.0] (Tỷ lệ mẫu dùng cho mỗi cây)
            \end{itemize}
        \item GridSearchCV thực hiện Cross-Validation (ví dụ: cv=3) cho mỗi tổ hợp tham số.
    \end{itemize}
    \textit{(Quá trình này có thể mất nhiều thời gian)}
\end{frame}

\begin{frame}{3.3 Mô tả điều kiện dừng}
    \textbf{Điều kiện dừng trong quá trình tìm kiếm (GridSearchCV):}
    \begin{itemize}
        \item \textbf{Duyệt hết không gian tham số:} Quá trình tìm kiếm dừng khi đã thử tất cả các tổ hợp siêu tham số được chỉ định trong \texttt{param\_grid}.
        \item \textbf{Cross-validation đầy đủ:} Mỗi tổ hợp tham số được đánh giá bằng k-fold cross-validation (ví dụ k=3) để đảm bảo độ tin cậy của điểm số đánh giá.
    \end{itemize}
    \vspace{1em}
     \textbf{Điều kiện dừng trong quá trình huấn luyện Gradient Boosting (cho một bộ tham số cụ thể):}
     \begin{itemize}
        \item \textbf{Số lượng cây cố định:} Mô hình dừng khi đạt đến số lượng cây đã chỉ định (\texttt{n\_estimators}).
        \item \textbf{Early stopping (nếu được bật):} Dừng sớm nếu hiệu suất trên tập validation không cải thiện sau một số vòng lặp nhất định (không dùng trong GridSearchCV mặc định).
        \item \textbf{Điều kiện dừng của các cây riêng lẻ:} Mỗi cây quyết định con dừng phân tách khi đạt đến độ sâu tối đa (\texttt{max\_depth}) hoặc khi số lượng mẫu tại nút nhỏ hơn \texttt{min\_samples\_split}.
     \end{itemize}
\end{frame}

\begin{frame}{3.4 Đánh giá tinh chỉnh: Trước & Sau}

    \begin{itemize}
        \item \textbf{1. Trước khi hiệu chỉnh (Mô hình mặc định trên CV tập huấn luyện):}
            \begin{itemize}
                 \item RMSE = 26,126.56 (±1199.20) USD
            \end{itemize}
        \item \textbf{2. Sau khi hiệu chỉnh (Kết quả tốt nhất từ GridSearchCV trên CV tập huấn luyện):}
             \begin{itemize}
                 \item RMSE = 16,168.27 USD
             \end{itemize}
        \item \textbf{3. Trên tập test (Mô hình tốt nhất sau GridSearchCV, đánh giá trên tập test riêng):}

             \end{itemize}

    \textbf{Nhận xét:}
    \begin{itemize}

        \item \textbf{Hiệu suất xuất sắc trên tập test:} RMSE trên tập test thậm chí còn thấp hơn một chút so với kết quả CV tốt nhất, cho thấy mô hình không bị overfitting và tổng quát hóa cực kỳ tốt trên dữ liệu mới.
    \end{itemize}
\end{frame}

\begin{frame}{3.5 Lựa chọn số đo đánh giá mô hình}
    Các số đo được sử dụng để đánh giá mô hình hồi quy cuối cùng:
    \begin{itemize}
        \item \textbf{MAE (Mean Absolute Error):}
            \begin{itemize}
                \item Đo lường sai số tuyệt đối trung bình.
                \item $MAE = \frac{1}{n} \sum_{i=1}^{n} |y_i - \hat{y}_i|$
                \item Cùng đơn vị với biến mục tiêu. Ít bị ảnh hưởng bởi outliers hơn RMSE.
            \end{itemize}
        \item \textbf{RMSE (Root Mean Squared Error):}
            \begin{itemize}
                 \item Đo lường căn bậc hai của sai số bình phương trung bình.
                 \item $RMSE = \sqrt{\frac{1}{n} \sum_{i=1}^{n} (y_i - \hat{y}_i)^2}$
                 \item Cùng đơn vị với biến mục tiêu. Nhấn mạnh các lỗi lớn.
            \end{itemize}
         \item \textbf{R² (R-squared):}
             \begin{itemize}
                 \item Chỉ số xác định mức độ biến thiên của biến mục tiêu được giải thích bởi mô hình.
                 \item $R^2 = 1 - \frac{\sum_{i=1}^{n} (y_i - \hat{y}_i)^2}{\sum_{i=1}^{n} (y_i - \bar{y})^2}$
                 \item Giá trị từ $-\infty$ đến 1. Càng gần 1 càng tốt.
             \end{itemize}
        \item \textbf{MAPE (Mean Absolute Percentage Error):}
             \begin{itemize}
                 \item Đo lường sai số phần trăm tuyệt đối trung bình.
                 \item $MAPE = \frac{1}{n} \sum_{i=1}^{n} |\frac{y_i - \hat{y}_i}{y_i}| \times 100\%$
                 \item Cho biết sai số tương đối, hữu ích khi so sánh trên các tập dữ liệu khác nhau. (Lưu ý: có thể không ổn định nếu $y_i$ gần 0).
             \end{itemize}
    \end{itemize}
\end{frame}

     \end{itemize}
\end{frame}

\begin{frame}{4.2 Diễn giải kết quả: Mẫu 3 (Bình dân)}
     \textbf{Mẫu 3: Nhà ở khu vực bình dân (Hialeah)}
      \begin{itemize}
        \item \textbf{Đặc điểm chính:} DT đất 6k, DT sống 1.4k sqft, 50 năm tuổi, chất lượng cấu trúc 2 (thấp), xa biển.
        \item Giá trị dự đoán bởi mô hình: \textbf{\$310,000}
        \item Giá trị thực tế tham khảo: \$325,000
        \item Sai số: \$310,000 - \$325,000 = -\$15,000 (\textbf{-4.6\%}) 
      \end{itemize}
     \textbf{Phân tích:}
      \begin{itemize}
        \item Mô hình dự đoán thấp hơn một chút so với giá thực tế (-4.6%).
        \item Sai số lớn hơn so với phân khúc cao cấp nhưng vẫn ở mức chấp nhận được.
        \item Phản ánh đúng các yếu tố làm giảm giá trị: nhà cũ (50 năm), chất lượng thấp (cấp 2), vị trí xa các yếu tố hấp dẫn (biển).
      \end{itemize}
\end{frame}

\begin{frame}{4.2 Diễn giải chung & 4.2.3 Giá trị thực tế}
     \textbf{Diễn giải chung về hiệu suất mô hình:}
     \begin{itemize}
        \item \textbf{Phân khúc giá và độ chính xác:} Mô hình hoạt động tốt nhất với phân khúc nhà cao cấp (sai số ~2.7%), độ chính xác giảm dần khi xuống phân khúc thấp hơn (trung bình ~4.2%, bình dân ~4.6%).
        \item \textbf{Ảnh hưởng của các yếu tố địa lý:} Vị trí (khoảng cách đến biển, mặt nước, trung tâm) có tác động mạnh mẽ, phản ánh đúng sự phân tầng của thị trường Miami.
        \item \textbf{Tác động của đặc điểm BĐS:} Tuổi nhà, chất lượng công trình, tỷ lệ diện tích sống/đất (\textit{nếu được tạo làm biến mới}) có mối tương quan mạnh với giá.
     \end{itemize}
      \textbf{Giá trị ứng dụng trong thực tế:}
      \begin{itemize}
        \item \textbf{Hỗ trợ định giá nhanh:} Cung cấp ước tính ban đầu đáng tin cậy (sai số trung bình 3-5%).
        \item \textbf{Phát hiện giá trị bất thường:} So sánh giá niêm yết với giá dự đoán có thể cảnh báo BĐS bị định giá quá cao/thấp.
        \item \textbf{Ứng dụng theo phân khúc:} Rất tin cậy cho phân khúc cao cấp. Cần kết hợp đánh giá chuyên gia cho phân khúc trung bình và thấp.
        \item \textbf{Giới hạn:} Không nắm bắt được yếu tố đặc thù (view đẹp, thiết kế đặc biệt, tình trạng nội thất...) có thể ảnh hưởng đến giá trị thực.
      \end{itemize}
\end{frame}

% --- PHẦN 5: KẾT LUẬN ---
\section{Kết luận}

\begin{frame}{5.1 Ưu điểm của cách tiếp cận}
    \begin{itemize}
        \item \textbf{Tiền xử lý dữ liệu toàn diện:} Xử lý outliers, chuẩn hóa, đảm bảo chất lượng dữ liệu đầu vào.
        \item \textbf{Tạo biến mới hiệu quả (nếu có):} Ví dụ: tỷ lệ diện tích sống/đất, giá/sqft, khoảng cách trung bình có thể cải thiện hiệu suất (được đề cập trong code ví dụ Flask).
        \item \textbf{So sánh đa dạng mô hình:} Thử nghiệm nhiều thuật toán (Linear, Ridge, Lasso, RF, GB) giúp tìm ra mô hình phù hợp nhất.
        \item \textbf{Tối ưu hóa siêu tham số hiệu quả:} Sử dụng GridSearchCV giúp cải thiện đáng kể hiệu suất so với mô hình mặc định.
        \item \textbf{Đánh giá đa chiều:} Sử dụng nhiều chỉ số (MAE, RMSE, R², MAPE) và phân tích trực quan (scatter plot, histogram sai số) cho cái nhìn toàn diện.
        \item \textbf{Phân tích độ quan trọng của đặc trưng (có thể có từ GB):} Cung cấp hiểu biết sâu sắc về các yếu tố ảnh hưởng đến giá nhà.
        \item \textbf{Khả năng ứng dụng thực tế:} Độ chính xác tốt (sai số 3-5% trên test) và có thể triển khai thành ứng dụng hỗ trợ định giá.
    \end{itemize}
\end{frame}

\begin{frame}{5.1 Nhược điểm của cách tiếp cận}
    \begin{itemize}
        \item \textbf{Xử lý không tối ưu các biến địa lý:} Sử dụng tọa độ (Latitude, Longitude) hoặc khoảng cách đơn thuần chưa nắm bắt hết đặc trưng phức tạp của các khu vực, lân cận trong Miami.
        \item \textbf{Độ chính xác không đồng đều theo phân khúc:} Hoạt động tốt với nhà cao cấp nhưng kém chính xác hơn với nhà giá thấp.
        \item \textbf{Chưa khai thác dữ liệu ngữ cảnh bên ngoài:} Thiếu thông tin về chất lượng trường học, tỷ lệ tội phạm, tiện ích xung quanh chi tiết (quán ăn, cửa hàng), giao thông công cộng...
        \item \textbf{Hạn chế trong việc nắm bắt yếu tố đặc biệt:} Khó đánh giá các yếu tố như view đẹp, thiết kế kiến trúc độc đáo, tình trạng sửa chữa/nâng cấp gần đây.
        \item \textbf{Thiếu xử lý dữ liệu theo thời gian:} Chưa tính đến xu hướng biến động giá theo thời gian (tháng, năm) và các yếu tố mùa vụ có thể ảnh hưởng đến giao dịch.
        \item \textbf{Chưa thử nghiệm các mô hình phức tạp hơn:} Các kỹ thuật như Deep Learning (Neural Networks) hay các phương pháp Ensemble tiên tiến khác (Stacking, Blending) chưa được khám phá.
    \end{itemize}
\end{frame}

\begin{frame}{5.2 Khả năng cải tiến trong tương lai (1/2)}
    \begin{itemize}
        \item \textbf{1. Tích hợp dữ liệu địa lý tốt hơn:}
            \begin{itemize}
                \item Áp dụng kỹ thuật clustering (phân cụm) K-Means hoặc DBSCAN dựa trên tọa độ để phân vùng các khu vực có đặc điểm giá tương đồng.
                \item Sử dụng Geohashing để tạo biến đại diện cho vị trí ô địa lý.
                \item Tạo các biến mới dựa trên thông tin khu phố, vùng lân cận (neighborhood features).
            \end{itemize}
        \item \textbf{2. Bổ sung dữ liệu ngữ cảnh phong phú:}
            \begin{itemize}
                \item Thu thập và tích hợp dữ liệu về xếp hạng trường học, thống kê tội phạm, mật độ tiện ích (nhà hàng, siêu thị), điểm số walkability...
                \item Dữ liệu về giao thông, kết nối công cộng.
            \end{itemize}
        \item \textbf{3. Áp dụng kỹ thuật mô hình hóa tiên tiến:}
            \begin{itemize}
                 \item Thử nghiệm các thuật toán gradient boosting nâng cao như XGBoost, LightGBM, CatBoost (thường hiệu quả hơn và nhanh hơn).
                 \item Áp dụng mô hình Deep Learning (ví dụ: Multi-layer Perceptron) nếu dữ liệu đủ lớn và phức tạp.
                 \item Xây dựng mô hình Ensemble kết hợp dự đoán từ nhiều mô hình khác nhau (Stacking, Blending).
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{5.2 Khả năng cải tiến trong tương lai (2/2)}
     \begin{itemize}
        \item \textbf{4. Tối ưu hóa mô hình theo phân khúc:}
            \begin{itemize}
                \item Xây dựng các mô hình riêng biệt cho từng phân khúc giá (cao cấp, trung bình, bình dân).
                \item Áp dụng kỹ thuật sample weighting để cân bằng hiệu suất giữa các phân khúc nếu dùng mô hình chung.
                \item Phân tích lỗi chi tiết theo từng phân khúc để xác định nguyên nhân sai số và cải thiện.
            \end{itemize}
        \item \textbf{5. Tích hợp phân tích dữ liệu theo thời gian:}
             \begin{itemize}
                 \item Xây dựng các biến thể hiện xu hướng thị trường theo thời gian (time trends).
                 \item Phân tích và mô hình hóa yếu tố mùa vụ, chu kỳ kinh tế.
                 \item Xây dựng mô hình không gian-thời gian (spatio-temporal models).
             \end{itemize}
        \item \textbf{6. Cải thiện trải nghiệm ứng dụng:}
             \begin{itemize}
                 \item Phát triển giao diện người dùng trực quan với bản đồ nhiệt giá, so sánh BĐS.
                 \item Tích hợp công cụ "What-if" để người dùng thấy tác động của việc thay đổi đặc điểm đến giá.
             \end{itemize}
         \item \textbf{7. Xây dựng hệ thống cập nhật liên tục (MLOps):}
            \begin{itemize}
                 \item Tự động thu thập dữ liệu giao dịch mới, tái huấn luyện mô hình định kỳ, theo dõi hiệu suất theo thời gian thực.
            \end{itemize}
          \item \textbf{8. Kết hợp với phân tích định tính:}
             \begin{itemize}
                  \item Tích hợp đánh giá từ chuyên gia BĐS.
                  \item Phân tích dữ liệu văn bản từ mô tả BĐS (NLP).
                  \item Tự động đánh giá hình ảnh BĐS bằng Computer Vision.
             \end{itemize}
     \end{itemize}
\end{frame}
